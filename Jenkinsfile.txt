pipeline {
    agent any
    stages {

        stage('Branch Master') {
            when {
                branch 'master'
            }

            steps {
                echo 'Building The Master Branch...'
                bat 'mvn clean compile'
            }

        }

        stage('Branch Dev') {
            when {
                branch 'dev'
            }

            steps {
                echo 'Building dev Branch...'
                bat 'mvn clean compile'
            }
        }

        stage('Branch Test') {
            when {
                branch 'test'
            }

            steps {
                echo 'Building test Branch...'
            }
        }


    }

}
